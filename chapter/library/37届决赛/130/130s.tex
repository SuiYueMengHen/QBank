            \label{eq:17.s130}

        \begin{equation}

            \label{eq:20.s130}

        \begin{align}
            \sqrt {d _ {1} ^ {2} + 2 d _ {1} K} &\leq \sqrt {d _ {2} ^ {2} + 2 d _ {2} K} + \sqrt {d _ {3} ^ {2} + 2 d _ {3} K} \label{eq:21.s130} \\

        \end{align}

        \begin{equation}
            d _ {1} = y, \quad d _ {2} = - (\sqrt {3} x + y - 1) / 2, \quad d _ {3} = (\sqrt {3} x - y + 1) / 2

        \end{equation}

        \end{equation}
